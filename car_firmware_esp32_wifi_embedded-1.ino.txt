#define SCREEN_WIDTH 128
#define SCREEN_HEIGHT 64
Adafruit_SSD1306 display(SCREEN_WIDTH, SCREEN_HEIGHT, &Wire, -1);

// --- MOTOR PINS ---
int enA = 25; int in1 = 26; int in2 = 27;
int enB = 33; int in3 = 23; int in4 = 13;

// --- SENSOR PINS ---
int trigPin = 32; int echoPin = 35;
int leftIR_Front = 34; int rightIR_Front = 39;
int leftIR_Back = 36;  int rightIR_Back = 4;

// --- EXTRAS ---
int servoPin = 18;
int buzzer = 19;
int lights = 17;
int ledPin = 2;
Servo scanServo;

// I2C for OLED: default SDA=21 / SCL=22

// --- SYSTEM VARIABLES ---
char currentCommand = 'S'; 
char carMode = 'M'; 

int manualSpeed = 240;
const int SPEED_MIN = 120;
const int SPEED_MAX = 255;
const int SPEED_STEP = 15;

bool usEnabled = true;
bool irEnabled = true;
bool lightsEnabled = false;
bool led13Enabled = false;
bool manualBuzzerOn = false;

// --- DISPLAY VARIABLES ---
const char* displayMode = "MANUAL";
const char* displayAction = "STOP";
const char* lastMode = "";
const char* lastAction = "";

unsigned long rotateStartTime = 0;
bool isRotating = false;

// --- NON-BLOCKING BUZZER STATE MACHINE ---
enum BuzzerState { BUZZ_IDLE, BUZZ_ON, BUZZ_GAP };
BuzzerState buzzerState = BUZZ_IDLE;
unsigned long buzzerTimer = 0;
int buzzerBeepsRemaining = 0;
const unsigned long BEEP_ON_MS = 150;
const unsigned long BEEP_GAP_MS = 100;

void startBeep(int count) {
  buzzerBeepsRemaining = count - 1; 
  digitalWrite(buzzer, HIGH);
  buzzerTimer = millis();
  buzzerState = BUZZ_ON;
}

void updateBuzzer() {
  if (manualBuzzerOn) {
    digitalWrite(buzzer, HIGH);
    return;
  }
  unsigned long now = millis();
  if (buzzerState == BUZZ_ON && now - buzzerTimer >= BEEP_ON_MS) {
    digitalWrite(buzzer, LOW);
    buzzerTimer = now;
    buzzerState = (buzzerBeepsRemaining > 0) ? BUZZ_GAP : BUZZ_IDLE;
  } else if (buzzerState == BUZZ_GAP && now - buzzerTimer >= BEEP_GAP_MS) {
    digitalWrite(buzzer, HIGH);
    buzzerTimer = now;
    buzzerBeepsRemaining--;
    buzzerState = BUZZ_ON;
  }
}

// --- WIFI-AWARE SMART DELAY ---
volatile bool interruptFlag = false;

bool smartDelay(unsigned long ms) {
  unsigned long start = millis();
  interruptFlag = false;
  while (millis() - start < ms) {
    updateBuzzer();
    server.handleClient();
    if (interruptFlag) return false;
  }
  return true;
}

// --- DEFERRED BANNER FLAGS ---
volatile bool pendingModeBanner = false;
const char* pendingBannerLine1 = "";
const char* pendingBannerLine2 = "";

volatile bool pendingConnectBanner = false;
bool wifiClientConnected = false;

void WiFiEventHandler(WiFiEvent_t event) {
  if (event == ARDUINO_EVENT_WIFI_AP_STACONNECTED) {
    if (!wifiClientConnected) {
      wifiClientConnected = true;
      pendingConnectBanner = true;
    }
  }
}

// --- HTTP HANDLERS ---
void handleCmd() {
  server.sendHeader("Access-Control-Allow-Origin", "*");

  if (!server.hasArg("c") || server.arg("c").length() == 0) {
    server.send(400, "text/plain", "missing c");
    return;
  }
  char incoming = server.arg("c").charAt(0);

  if (incoming == 'A') {
    carMode = 'A'; displayMode = "AUTO"; stopCar(); startBeep(1);
    pendingBannerLine1 = "AUTOMATIC"; pendingBannerLine2 = "MODE"; pendingModeBanner = true;
    interruptFlag = true;
  }
  else if (incoming == 'I') {
    carMode = 'I'; displayMode = "LINE"; stopCar(); startBeep(1);
    pendingBannerLine1 = "LINE"; pendingBannerLine2 = "MODE"; pendingModeBanner = true;
    interruptFlag = true;
  }
  else if (incoming == 'M') {
    carMode = 'M'; displayMode = "MANUAL"; stopCar(); startBeep(1);
    pendingBannerLine1 = "MANUAL"; pendingBannerLine2 = "MODE"; pendingModeBanner = true;
    interruptFlag = true;
  }
  else if (incoming == 'X') { usEnabled = !usEnabled; startBeep(usEnabled ? 1 : 2); }
  else if (incoming == 'Y') { irEnabled = !irEnabled; startBeep(irEnabled ? 1 : 2); }
  else if (incoming == 'H') {
    lightsEnabled = !lightsEnabled;
    digitalWrite(lights, lightsEnabled ? HIGH : LOW);
    startBeep(lightsEnabled ? 1 : 2);
  }
  else if (incoming == 'P') {
    led13Enabled = !led13Enabled;
    digitalWrite(ledPin, led13Enabled ? HIGH : LOW);
    startBeep(led13Enabled ? 1 : 2);
  }
  else if (incoming == 'Z') {
    manualBuzzerOn = !manualBuzzerOn;
    if (!manualBuzzerOn) { digitalWrite(buzzer, LOW); buzzerState = BUZZ_IDLE; }
  }
  else if (incoming == 'U') { manualSpeed = min(manualSpeed + SPEED_STEP, SPEED_MAX); startBeep(1); }
  else if (incoming == 'D') { manualSpeed = max(manualSpeed - SPEED_STEP, SPEED_MIN); startBeep(1); }
  else if (incoming == 'F' || incoming == 'B' || incoming == 'L' || incoming == 'R' || incoming == 'S') {
    if (carMode == 'M') currentCommand = incoming;
    if (incoming == 'S') interruptFlag = true;
  }

  server.send(200, "text/plain", "OK");
}

void handlePing() {
  server.sendHeader("Access-Control-Allow-Origin", "*");
  server.send(200, "text/plain", "OK");
}

void handleRoot() {
  server.send_P(200, "text/html", INDEX_HTML);
}

void handleNotFound() {
  server.sendHeader("Access-Control-Allow-Origin", "*");
  server.send(404, "text/plain", "Not found");
}

void setup() {
  Serial.begin(115200);

  pinMode(buzzer, OUTPUT);
  digitalWrite(buzzer, HIGH); delay(150); digitalWrite(buzzer, LOW);

  if (!display.begin(SSD1306_SWITCHCAPVCC, 0x3C)) {
    Serial.println(F("SSD1306 allocation failed"));
  }

  showBanner("HELLO!", "WAKING UP...");
  delay(1200);
  faceWakeUp();
  delay(200);
  faceSmile();
  delay(1200);

  pinMode(enA, OUTPUT); pinMode(enB, OUTPUT);
  pinMode(in1, OUTPUT); pinMode(in2, OUTPUT);
  pinMode(in3, OUTPUT); pinMode(in4, OUTPUT);

  pinMode(trigPin, OUTPUT); pinMode(echoPin, INPUT);
  pinMode(leftIR_Front, INPUT); pinMode(rightIR_Front, INPUT);
  pinMode(leftIR_Back, INPUT);  pinMode(rightIR_Back, INPUT);

  pinMode(lights, OUTPUT); digitalWrite(lights, LOW);
  pinMode(ledPin, OUTPUT); digitalWrite(ledPin, LOW);

  scanServo.setPeriodHertz(50);
  scanServo.attach(servoPin);
  scanServo.write(90);

  WiFi.onEvent(WiFiEventHandler);
  WiFi.mode(WIFI_AP);
  WiFi.softAP(AP_SSID, AP_PASSWORD);
  Serial.print("Car WiFi IP: ");
  Serial.println(WiFi.softAPIP());

  server.on("/", HTTP_GET, handleRoot);
  server.on("/cmd", HTTP_GET, handleCmd);
  server.on("/ping", HTTP_GET, handlePing);
  server.onNotFound(handleNotFound);
  server.begin();
}

void loop() {
  updateBuzzer();
  server.handleClient();

  if (pendingConnectBanner) {
    pendingConnectBanner = false;
    startBeep(1);
    showBanner("CONNECTED", "TO WIFI");
    smartDelay(1000);
    showBanner("MANUAL", "MODE");
    smartDelay(1000);
    updateScreenForce();
  }

  if (pendingModeBanner) {
    pendingModeBanner = false;
    showBanner(pendingBannerLine1, pendingBannerLine2);
    smartDelay(1000);
    updateScreenForce();
  }

  // ==========================================
  // MODE 'M': MANUAL DRIVING
  // ==========================================
  if (carMode == 'M') {
    bool canMoveForward = true;
    bool canMoveBackward = true;

    if (currentCommand == 'L' || currentCommand == 'R') {
      if (!isRotating) {
        isRotating = true;
        rotateStartTime = millis();
      } else if (millis() - rotateStartTime >= 6000) {
        stopCar(); currentCommand = 'S'; displayAction = "STOP";
        showAngryFace();
        smartDelay(4000);
        isRotating = false; lastAction = "";
      }
    } else { isRotating = false; }

    int distance = 999;
    int fLeft = LOW, fRight = LOW, bLeft = LOW, bRight = LOW;

    if (usEnabled) { distance = getDistance(); }
    if (irEnabled) {
      fLeft = digitalRead(leftIR_Front); fRight = digitalRead(rightIR_Front);
      bLeft = digitalRead(leftIR_Back);  bRight = digitalRead(rightIR_Back);
    }

    if ((irEnabled && (fLeft == HIGH || fRight == HIGH)) || (usEnabled && distance > 0 && distance <= 15)) {
      if (currentCommand == 'F') {
        stopCar(); startBeep(1);
        displayAction = "CLIFF/OBS!"; updateScreenForce();
        driveBackward(255);
        smartDelay(350);
        stopCar(); currentCommand = 'S';
      }
      canMoveForward = false;
    }

    if (irEnabled && (bLeft == HIGH || bRight == HIGH)) {
      if (currentCommand == 'B') {
        stopCar(); startBeep(1);
        displayAction = "CLIFF!"; updateScreenForce();
        driveForward(255);
        smartDelay(350);
        stopCar(); currentCommand = 'S';
      }
      canMoveBackward = false;
    }

    int fwdSpd  = constrain(manualSpeed,      0, 255);
    int bwdSpd  = constrain(manualSpeed - 10, 0, 255);
    int turnSpd = constrain(manualSpeed - 40, 0, 255);

    if (currentCommand == 'F' && canMoveForward) { driveForward(fwdSpd); displayAction = "FORWARD"; }
    else if (currentCommand == 'B' && canMoveBackward) { driveBackward(bwdSpd); displayAction = "REVERSE"; }
    else if (currentCommand == 'L') { driveLeft(turnSpd); displayAction = "LEFT"; }
    else if (currentCommand == 'R') { driveRight(turnSpd); displayAction = "RIGHT"; }
    else if (currentCommand == 'S' || (!canMoveForward && currentCommand == 'F') || (!canMoveBackward && currentCommand == 'B')) {
      stopCar(); displayAction = "STOP";
    }
  }

  // ==========================================
  // MODE 'A': AUTONOMOUS AVOIDANCE
  // ==========================================
  else if (carMode == 'A') {
    int fLeft = LOW, fRight = LOW, distance = 999;

    if (irEnabled) {
      fLeft = digitalRead(leftIR_Front);
      fRight = digitalRead(rightIR_Front);
    }

    if (irEnabled && (fLeft == HIGH || fRight == HIGH)) {
        stopCar();
        displayAction = "CLIFF!"; updateScreenForce();
        driveBackward(255);
        if (!smartDelay(400)) return;
        driveRight(200);
        if (!smartDelay(600)) return;
        stopCar();
    }
    else {
      if (usEnabled) { distance = getDistance(); }

      if (usEnabled && distance > 0 && distance <= 25) {
        stopCar();
        displayAction = "SCANNING"; updateScreenForce();

        if (!smartDelay(200)) return;
        scanServo.write(180);
        faceLookLeft();
        if (!smartDelay(400)) return;
        int leftDist = getDistance();

        scanServo.write(0);
        faceLookRight();
        if (!smartDelay(600)) return;
        int rightDist = getDistance();

        scanServo.write(90);
        faceNormal();
        if (!smartDelay(400)) return;

        if (leftDist > rightDist && leftDist > 25) {
          displayAction = "TURN L"; updateScreenForce();
          driveLeft(200); if (!smartDelay(400)) return;
        } else if (rightDist > leftDist && rightDist > 25) {
          displayAction = "TURN R"; updateScreenForce();
          driveRight(200); if (!smartDelay(400)) return;
        } else {
          displayAction = "TURN AROUND"; updateScreenForce();
          driveBackward(200); if (!smartDelay(500)) return;
          driveRight(200); if (!smartDelay(800)) return;
        }
        stopCar();
      } else {
        driveForward(150);
        displayAction = "FORWARD";
      }
    }
  }

  // ==========================================
  // MODE 'I': LINE FOLLOWER
  // ==========================================
  else if (carMode == 'I') {
    if (irEnabled) {
      int leftVal = digitalRead(leftIR_Front);
      int rightVal = digitalRead(rightIR_Front);
      int lineSpeed = 160;

      if (leftVal == LOW && rightVal == LOW) { driveForward(lineSpeed); displayAction = "FORWARD"; }
      else if (leftVal == HIGH && rightVal == LOW) { driveLeft(lineSpeed + 60); displayAction = "LEFT"; }
      else if (leftVal == LOW && rightVal == HIGH) { driveRight(lineSpeed + 60); displayAction = "RIGHT"; }
      else { stopCar(); displayAction = "STOP"; }
    } else {
      stopCar();
      displayAction = "IR IS OFF!";
    }
  }

  if (strcmp(displayMode, lastMode) != 0 || strcmp(displayAction, lastAction) != 0) {
    updateScreenForce();
  }
}

void drawEyes(int w, int h, int xShift, int yShift, int radius) {
  display.clearDisplay();
  int gap = 14;
  int totalW = w * 2 + gap;
  int leftX = (SCREEN_WIDTH - totalW) / 2 + xShift;
  int rightX = leftX + w + gap;
  int y = (SCREEN_HEIGHT - h) / 2 + yShift;
  if (radius > h / 2) radius = h / 2;
  if (radius > w / 2) radius = w / 2;
  display.fillRoundRect(leftX, y, w, h, radius, WHITE);
  display.fillRoundRect(rightX, y, w, h, radius, WHITE);
  display.display();
}

void showBanner(const char* line1, const char* line2) {
  display.clearDisplay();
  display.setTextColor(WHITE);

  display.setTextSize(2);
  int16_t x1, y1; uint16_t w1, h1;
  display.getTextBounds(line1, 0, 0, &x1, &y1, &w1, &h1);
  display.setCursor((SCREEN_WIDTH - w1) / 2, 14);
  display.print(line1);

  display.setTextSize(1);
  int16_t x2, y2; uint16_t w2, h2;
  display.getTextBounds(line2, 0, 0, &x2, &y2, &w2, &h2);
  display.setCursor((SCREEN_WIDTH - w2) / 2, 40);
  display.print(line2);

  display.display();
}

void updateScreenForce() {
  if (strcmp(displayAction, "FORWARD") == 0) faceExcited();
  else if (strcmp(displayAction, "REVERSE") == 0) faceNormal();
  else if (strcmp(displayAction, "LEFT") == 0) faceLookLeft();
  else if (strcmp(displayAction, "RIGHT") == 0) faceLookRight();
  else if (strcmp(displayAction, "STOP") == 0) faceSmile();
  else if (strcmp(displayAction, "CLIFF/OBS!") == 0) faceFear();
  else if (strcmp(displayAction, "CLIFF!") == 0) faceFear();
  else if (strcmp(displayAction, "SCANNING") == 0) faceNormal();
  else if (strcmp(displayAction, "TURN L") == 0) faceLookLeft();
  else if (strcmp(displayAction, "TURN R") == 0) faceLookRight();
  else if (strcmp(displayAction, "TURN AROUND") == 0) faceSurprised();
  else faceSmile();

  lastMode = displayMode;
  lastAction = displayAction;
}

// --- SENSOR FUNCTIONS ---
int getDistance() {
  digitalWrite(trigPin, LOW); delayMicroseconds(2);
  digitalWrite(trigPin, HIGH); delayMicroseconds(10);
  digitalWrite(trigPin, LOW);
  long duration = pulseIn(echoPin, HIGH, 25000);
  int cm = duration * 0.034 / 2;
  return (cm == 0) ? 999 : cm;
}

// --- MOTOR FUNCTIONS ---
void driveForward(int speed) {
  digitalWrite(in1, HIGH); digitalWrite(in2, LOW);
  digitalWrite(in3, HIGH); digitalWrite(in4, LOW);
  analogWrite(enA, speed); analogWrite(enB, speed);
}
void driveBackward(int speed) {
  digitalWrite(in1, LOW);  digitalWrite(in2, HIGH);
  digitalWrite(in3, LOW);  digitalWrite(in4, HIGH);
  analogWrite(enA, speed); analogWrite(enB, speed);
}
void driveLeft(int speed) {
  digitalWrite(in1, LOW);  digitalWrite(in2, HIGH);
  digitalWrite(in3, HIGH); digitalWrite(in4, LOW);
  analogWrite(enA, speed); analogWrite(enB, speed);
}
void driveRight(int speed) {
  digitalWrite(in1, HIGH); digitalWrite(in2, LOW);
  digitalWrite(in3, LOW);  digitalWrite(in4, HIGH);
  analogWrite(enA, speed); analogWrite(enB, speed);
}
void stopCar() {
  analogWrite(enA, 0);   analogWrite(enB, 0);
  digitalWrite(in1, LOW); digitalWrite(in2, LOW);
  digitalWrite(in3, LOW); digitalWrite(in4, LOW);
}
